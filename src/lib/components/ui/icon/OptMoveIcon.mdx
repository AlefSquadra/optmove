# Icones OptMove

Esta seção exibe todos os ícones disponíveis no sistema OptMove.

## Lista de icones

{/* Criando um componente funcional para o filtro */}
{(() => {
const [filter, setFilter] = useState('');

return (

<>
<input
type="text"
placeholder="Filtrar ícones por nome..."
onChange={(e) => setFilter(e.target.value)}
style={{ marginBottom: '20px', padding: '8px', borderRadius: '4px', border: '1px solid #ccc' }}
/>

      <div style={{ display: 'grid', gridTemplateColumns: 'repeat(auto-fill, minmax(150px, 1fr))', gap: '20px' }}>
        {Object.entries(Icons)
          .filter(([name]) => name.toLowerCase().includes(filter.toLowerCase()))
          .map(([name, IconComponent]) => (
            <div key={name} style={{ textAlign: 'center', display: 'flex', alignItems: 'center', flexDirection: 'column' }}>
              <OptMoveIcon name={name} style={{ width: '50px', height: '50px' }} />
              <p style={{ wordBreak: 'break-all', marginTop: '10px' }}>{name}</p>
            </div>
          ))}
      </div>
    </>

);
})()}
